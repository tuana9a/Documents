# redirect stdout

```bash
# VD: ghi đè toàn bộ nội dung file gemdino.txt
# nếu chưa có file sẽ tạo file mới
echo hi Im Nguyen Quynh Anh > gemdino.txt

# VD: ghi tiếp vào file "gemdino.txt" dòng text trước dấu ">>"
# không ghi đè mà là ghi tiếp !!!
echo hi Im Nguyen Quynh Anh >> gemdino.txt
```
